function getUsersRows($result)
{
   $tableRows = "";
   foreach ($result as $row)
   {
      $tableRows .= "<tr>\n" .
       "<td>" . $row['user']   . "</td>" .
       "<td>" . $row['lastname'] . "</td>" .
       "<td>" . $row['firstname'] . "</td>" .
       "<td>" . "$" . number_format($row['telephone'], 2)  . "</td>" .
       "</tr>";
   }
   return $tableRows;
}
